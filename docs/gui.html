 519    def load_config_file(self) -> None:
 520        """
 521        Open a file dialog to select a configuration file and load the settings.
 522        """
 523        file_path, _ = QFileDialog.getOpenFileName(
 524            self, "Select Config File", "", "Config Files (*.env *.txt);;All Files (*)"
 525        )
 526        if file_path:
 527            try:
 528                with open(file_path, 'r') as f:
 529                    data = {}
 530                    for line in f:
 531                        line = line.strip()
 532                        if not line or line.startswith('#'):
 533                            continue  # Skip empty lines and comments
 534                        if '=' in line:
 535                            key, value = line.split('=', 1)
 536                            key = key.strip()
 537                            value = value.strip()
 538                            data[key] = value
 539                    # Update the settings entries with loaded data
 540                    for key, widget in self.settings_entries.items():
 541                        if key in data:
 542                            value = data[key]
 543                            if isinstance(widget, QLineEdit):
 544                                widget.setText(value)
 545                            elif isinstance(widget, QComboBox):
 546                                index = widget.findText(value, Qt.MatchFlag.MatchFixedString)
 547                                if index >= 0:
 548                                    widget.setCurrentIndex(index)
 549                QMessageBox.information(self, "Success", f"Settings loaded from {file_path}")
 550            except Exception as e:
 551                QMessageBox.warning(self, "Error", f"Error loading config file: {e}")

 584        for key, widget in self.settings_entries.items():
 585            if isinstance(widget, QLineEdit):

 587            elif isinstance(widget, QComboBox):

 579    def save_settings(self) -> None:
 580        """
 581        Save the settings to 'config.env' after validating the input fields.
 582        """
 583        data = {}
 584        errors = []
 585        for key, widget in self.settings_entries.items():
 586            if isinstance(widget, QLineEdit):
 587                value = widget.text()
 588            elif isinstance(widget, QComboBox):
 589                value = widget.currentText()
 590            else:
 591                continue
 592            # Get min_length and max_length
 593            min_length = getattr(widget, 'min_length', 0)
 594            max_length = getattr(widget, 'max_length', None)
 595            if min_length and len(value) < min_length:
 596                errors.append(f"{key} must be at least {min_length} characters.")
 597            if max_length and len(value) > max_length:
 598                errors.append(f"{key} must be at most {max_length} characters.")
 599            # Specific validation for NOTIFY
 600            if key == 'NOTIFY' and value.upper() not in ['TRUE', 'FALSE']:
 601                errors.append(f"{key} must be TRUE or FALSE.")
 602            # Specific validation for FLAG
 603            if key == 'FLAG' and value:
 604                ext = os.path.splitext(value)[1].lower()
 605                if ext not in ['.svg', '.png', '.jpeg', '.jpg', '.gif']:
 606                    errors.append(f"{key} must be one of the following types: SVG, PNG, JPEG, or GIF.")
 607            data[key] = value
 608        if errors:
 609            QMessageBox.warning(self, "Validation Error", "\n".join(errors))
 610            return
 611        # Save data to config.env
 612        config_path = os.path.join(os.getcwd(), 'config.env')
 613        try:
 614            with open(config_path, 'w') as f:
 615                for key, value in data.items():
 616                    f.write(f"{key}={value}\n")
 617            QMessageBox.information(self, "Success", f"Settings saved to {config_path}")
 618        except Exception as e:
 619            QMessageBox.warning(self, "Error", f"Error saving settings: {e}")
