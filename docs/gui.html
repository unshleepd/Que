gui

This script creates a graphical user interface (GUI) for interacting with the que.py script. It allows users to process nations in the NationStates game through a user-friendly interface, providing options to change settings, change flags, move to regions, and place bids on cards.

 90    def update_buttons(self):
 91        """
 92        Updates the button colors based on the current state.
 93        """
 94        if self.variable.get():
 95            # If the switch is 'On', set the 'On' button to green
 96            self.on_button.config(bg="green", fg="white")
 97            self.off_button.config(bg="gray", fg="black")

 99            # If the switch is 'Off', set the 'Off' button to red
100            self.on_button.config(bg="gray", fg="black")

163class TextHandler(logging.Handler):
164    """
165    Logging handler that directs log messages to a Tkinter Text widget.
166    
167    Attributes:
168        text_widget (tk.Text): The text widget where logs will be displayed.
169    """
170    def __init__(self, text_widget):
171        """
172        Initialize the TextHandler with a reference to a text widget.
173        
174        Parameters:
175            text_widget (tk.Text): The widget where log messages will be displayed.
176        """
177        logging.Handler.__init__(self)
178        self.text_widget = text_widget  # Reference to the text widget
179
180    def emit(self, record):
181        """
182        Emit a record to the text widget.
183        
184        Parameters:
185            record (logging.LogRecord): The log record to be emitted.
186        """
187        msg = self.format(record)  # Format the log message
188        # Ensure thread-safe GUI updates using after()
189        log_window.after(0, self.write_message, msg)  # Schedule the message to be written in the main thread
190
191    def write_message(self, msg):
192        """
193        Write a message to the text widget in a thread-safe manner.
194        
195        Parameters:
196            msg (str): The message to write to the text widget.
197        """
198        self.text_widget.config(state='normal')  # Make the text widget editable
199        self.text_widget.insert(tk.END, msg + '\n')  # Insert the log message at the end
200        self.text_widget.see(tk.END)  # Scroll to the end to show the new message
201        self.text_widget.config(state='disabled')  # Disable editing again

203def setup_logging(log_filename="que_log.txt"):
204    """
205    Sets up logging to both a file and the GUI log window.
206    
207    Parameters:
208        log_filename (str): The name of the log file.
209    """
210    logger = logging.getLogger()  # Get the root logger
211    logger.setLevel(logging.INFO)  # Set the logging level to INFO
212    formatter = logging.Formatter('%(asctime)s %(message)s', datefmt='%I:%M:%S %p')  # Define the log message format
213
214    # File handler to write logs to a file
215    file_handler = logging.FileHandler(log_filename)  # Create a handler for the log file
216    file_handler.setFormatter(formatter)  # Set the formatter for the file handler
217    logger.addHandler(file_handler)  # Add the file handler to the logger
218
219    # GUI handler to write logs to the GUI
220    text_handler = TextHandler(log_window)  # Create a handler for the GUI log window
221    text_handler.setFormatter(formatter)  # Set the formatter for the text handler
222    logger.addHandler(text_handler)  # Add the text handler to the logger

227def start_script():
228    """
229    Starts the processing script in a separate thread.
230    """
231    global script_thread  # Access the global variable
232    if script_thread and script_thread.is_alive():
233        # If the script is already running, show a warning message
234        messagebox.showwarning("Warning", "The script is already running!")
235        return
236
237    if not selected_file:
238        # If no file has been selected, prompt the user to select one
239        messagebox.showwarning("Warning", "Please select a nations file before starting the script.")
240        return
241
242    # Disable the Start button to prevent multiple runs
243    start_button.config(state=tk.DISABLED)
244    # Retrieve the values from the switches
245    change_settings = change_settings_var.get()  # Get the state of 'Change Settings' switch
246    change_flag = change_flag_var.get()          # Get the state of 'Change Flag' switch
247    move_region = move_region_var.get()          # Get the state of 'Move to Region' switch
248    place_bids = place_bids_var.get()            # Get the state of 'Place Bids' switch
249
250    # Start the script in a new thread to keep the GUI responsive
251    script_thread = threading.Thread(
252        target=run_script,
253        args=(change_settings, change_flag, move_region, place_bids)
254    )
255    script_thread.start()  # Start the new thread
256
257def run_script(change_settings, change_flag, move_region, place_bids):
258    """
259    Runs the main processing script with the given parameters.
260    
261    Parameters:
262        change_settings (bool): Whether to change nation settings.
263        change_flag (bool): Whether to change the nation's flag.
264        move_region (bool): Whether to move the nation to a target region.
265        place_bids (bool): Whether to place bids on cards.
266    """
267    global script_thread  # Access the global variable
268    try:
269        logging.info("Starting script...")  # Log the start of the script
270        # Retrieve environment variables
271        env_vars = get_env_vars()  # Get environment variables needed for processing
272        # Initialize the NationStates session
273        session = NSSession("Que", "2.2.0", "Unshleepd", env_vars['UA'])  # Create a session object
274        # Read the nations from the selected file
275        with open(selected_file, "r") as q:
276            pups = q.readlines()  # Read all lines (nations) from the file
277        logging.info("Processing nations...")  # Log that nation processing is starting
278        # Process the nations with the specified options
279        process_nations(
280            session,
281            pups,
282            env_vars,
283            change_settings,
284            change_flag,
285            move_region,
286            place_bids
287        )
288    except Exception as e:
289        error_message = f"An error occurred: {e}"  # Prepare the error message
290        logging.error(error_message)  # Log the error
291        # Display error dialog in the GUI
292        root.after(0, lambda: messagebox.showerror("Error", error_message))  # Show error message box
293    else:
294        # If successful, call script_completed in the main thread
295        root.after(0, script_completed)
296    finally:
297        # Re-enable the "Start" button after completion
298        root.after(0, lambda: start_button.config(state=tk.NORMAL))
299        script_thread = None  # Reset the script thread variable

301def script_completed():
302    """
303    Called when the script has completed successfully.
304    """
305    logging.info("Process completed successfully.")  # Log completion message
306    # Display completion dialog to the user
307    messagebox.showinfo("Completion", "The process has completed successfully.")  # Show info message box

309def on_closing():
310    """
311    Handles the window closing event, ensuring the script thread is terminated properly.
312    """
313    if script_thread and script_thread.is_alive():
314        # If the script is running, confirm if the user wants to quit
315        if messagebox.askokcancel("Quit", "The script is still running. Do you want to quit?"):
316            root.destroy()  # Destroy the window and exit the application
317    else:
318        root.destroy()  # Destroy the window and exit the application
319
320# Set the protocol for handling window close events
321root.protocol("WM_DELETE_WINDOW", on_closing)  # Bind the on_closing function to window close event

259    """

262    Parameters:

275        # Read the nations from the selected file
276        with open(selected_file, "r") as q:
